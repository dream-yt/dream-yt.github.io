こんな関数で取得できる

const f = (file: File) => {
  const reader = new FileReader();
  const promise = new Promise((resolve, reject) => {
    reader.onerror = e => reject(e);
    reader.onabort = e => reject(e);
    reader.onload = e => resolve(e.target?.result);
  });
  reader.readAsDataURL(file);
  return promise;
};
